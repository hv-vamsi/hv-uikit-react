import { promises as fs } from "fs";

import { Layout } from "../../components/Layout"

import { getComponentData } from "../../utils.tsx"

export const getStaticProps = async ({ params }) => {

    const meta = await getComponentData("Tabs")

    return fs.readFile(`./../../samples/Tabs/FullWidth.tsx`, "utf8").then((values) => {
        return {
            props: {
                ssg: {
                    meta: meta,
                    demo: values
                },
            },
        }
    });

}
